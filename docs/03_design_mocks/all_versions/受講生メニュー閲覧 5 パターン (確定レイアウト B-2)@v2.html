    ${notes.map(n => `

${cycles.length > 1 ? `

`).join("")}

${escape(firstCycle["シート名"].replace(/【.*?】/g, "").trim())}

${samples.map((s, i) => renderPattern(s, i) + (i < samples.length - 1 ? '
